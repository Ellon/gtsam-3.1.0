99  static PinholeCamera Lookat(const Point3& eye, const Point3& target,
100  const Point3& upVector, const Calibration& K = Calibration()) {
101  Point3 zc = target - eye;
102  zc = zc / zc.norm();
103  Point3 xc = (-upVector).cross(zc); // minus upVector since yc is pointing down

272  inline static Point2 project_to_camera(const Point3& P,
273  boost::optional<Matrix&> Dpoint = boost::none) {
274 #ifdef GTSAM_THROW_CHEIRALITY_EXCEPTION
275  if (P.z() <= 0)
276  throw CheiralityException();
277 #endif
278  if (Dpoint) {
279  double d = 1.0 / P.z(), d2 = d * d;
280  *Dpoint = (Matrix(2, 3) << d, 0.0, -P.x() * d2, 0.0, d, -P.y() * d2);
281  }
282  return Point2(P.x() / P.z(), P.y() / P.z());
283  }

298  inline Point2 project(
299  const Point3& pw, //
300  boost::optional<Matrix&> Dpose = boost::none,
301  boost::optional<Matrix&> Dpoint = boost::none,
302  boost::optional<Matrix&> Dcal = boost::none) const {
303 
304  // Transform to camera coordinates and check cheirality
305  const Point3 pc = pose_.transform_to(pw);
306 
307  // Project to normalized image coordinates
308  const Point2 pn = project_to_camera(pc);
309 
310  if (Dpose || Dpoint) {
311  const double z = pc.z(), d = 1.0 / z;
312 
313  // uncalibration
314  Matrix Dpi_pn(2, 2);
315  const Point2 pi = K_.uncalibrate(pn, Dcal, Dpi_pn);
316 
317  // chain the Jacobian matrices
318  if (Dpose) {
319  Dpose->resize(2, 6);
320  calculateDpose(pn, d, Dpi_pn, *Dpose);
321  }
322  if (Dpoint) {
323  Dpoint->resize(2, 3);
324  calculateDpoint(pn, d, pose_.rotation().matrix(), Dpi_pn, *Dpoint);
325  }
326  return pi;
327  } else

337  inline Point2 projectPointAtInfinity(
338  const Point3& pw, //
339  boost::optional<Matrix&> Dpose = boost::none,
340  boost::optional<Matrix&> Dpoint = boost::none,
341  boost::optional<Matrix&> Dcal = boost::none) const {
342 
343  if (!Dpose && !Dpoint && !Dcal) {
344  const Point3 pc = pose_.rotation().unrotate(pw); // get direction in camera frame (translation does not matter)
345  const Point2 pn = project_to_camera(pc); // project the point to the camera
346  return K_.uncalibrate(pn);
347  }
348 
349  // world to camera coordinate
350  Matrix Dpc_rot /* 3*3 */, Dpc_point /* 3*3 */;
351  const Point3 pc = pose_.rotation().unrotate(pw, Dpc_rot, Dpc_point);
352 
353  Matrix Dpc_pose = Matrix::Zero(3, 6);
354  Dpc_pose.block(0, 0, 3, 3) = Dpc_rot;
355 
356  // camera to normalized image coordinate
357  Matrix Dpn_pc; // 2*3
358  const Point2 pn = project_to_camera(pc, Dpn_pc);
359 
360  // uncalibration
361  Matrix Dpi_pn; // 2*2
362  const Point2 pi = K_.uncalibrate(pn, Dcal, Dpi_pn);
363 
364  // chain the Jacobian matrices
365  const Matrix Dpi_pc = Dpi_pn * Dpn_pc;
366  if (Dpose)
367  *Dpose = Dpi_pc * Dpc_pose;
368  if (Dpoint)
369  *Dpoint = (Dpi_pc * Dpc_point).block(0, 0, 2, 2); // only 2dof are important for the point (direction-only)
370  return pi;
371  }

518  template<typename Derived>
519  static void calculateDpose(const Point2& pn, double d, const Matrix& Dpi_pn,
520  Eigen::MatrixBase<Derived> const & Dpose) {
521  // optimized version of derivatives, see CalibratedCamera.nb
522  const double u = pn.x(), v = pn.y();
523  double uv = u * v, uu = u * u, vv = v * v;
524  Eigen::Matrix<double, 2, 6> Dpn_pose;
525  Dpn_pose << uv, -1 - uu, v, -d, 0, d * u, 1 + vv, -uv, -u, 0, -d, d * v;
526  assert(Dpose.rows()==2 && Dpose.cols()==6);
527  const_cast<Eigen::MatrixBase<Derived>&>(Dpose) = //
528  Dpi_pn.block<2, 2>(0, 0) * Dpn_pose;
529  }

539  template<typename Derived>
540  static void calculateDpoint(const Point2& pn, double d, const Matrix& R,
541  const Matrix& Dpi_pn, Eigen::MatrixBase<Derived> const & Dpoint) {
542  // optimized version of derivatives, see CalibratedCamera.nb
543  const double u = pn.x(), v = pn.y();
544  Eigen::Matrix<double, 2, 3> Dpn_point;
545  Dpn_point << //
546  R(0, 0) - u * R(0, 2), R(1, 0) - u * R(1, 2), R(2, 0) - u * R(2, 2), //
547  R(0, 1) - v * R(0, 2), R(1, 1) - v * R(1, 2), R(2, 1) - v * R(2, 2);
548  Dpn_point *= d;
549  assert(Dpoint.rows()==2 && Dpoint.cols()==3);
550  const_cast<Eigen::MatrixBase<Derived>&>(Dpoint) = //
551  Dpi_pn.block<2, 2>(0, 0) * Dpn_point;
552  }
