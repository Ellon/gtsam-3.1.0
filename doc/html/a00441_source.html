37  template<typename KEYS>
38  JacobianFactor::JacobianFactor(
39  const KEYS& keys, const VerticalBlockMatrix& augmentedMatrix, const SharedDiagonal& model) :
40  Base(keys), Ab_(augmentedMatrix)
41  {
42  // Check noise model dimension
43  if(model && (DenseIndex)model->dim() != augmentedMatrix.rows())
44  throw InvalidNoiseModel(augmentedMatrix.rows(), model->dim());
45 
46  // Check number of variables
47  if((DenseIndex)Base::keys_.size() != augmentedMatrix.nBlocks() - 1)
48  throw std::invalid_argument(
49  "Error in JacobianFactor constructor input. Number of provided keys plus\n"
50  "one for the RHS vector must equal the number of provided matrix blocks.");
51 
52  // Check RHS dimension
53  if(augmentedMatrix(augmentedMatrix.nBlocks() - 1).cols() != 1)
54  throw std::invalid_argument(
55  "Error in JacobianFactor constructor input. The last provided matrix block\n"
56  "must be the RHS vector, but the last provided block had more than one column.");
57 
58  // Take noise model
59  model_ = model;
60  }

70  template<typename TERMS>
71  void JacobianFactor::fillTerms(const TERMS& terms, const Vector& b, const SharedDiagonal& noiseModel)
72  {
73  using boost::adaptors::transformed;
74  namespace br = boost::range;
75 
76  // Check noise model dimension
77  if(noiseModel && (DenseIndex)noiseModel->dim() != b.size())
78  throw InvalidNoiseModel(b.size(), noiseModel->dim());
79 
80  // Resize base class key vector
81  Base::keys_.resize(terms.size());
82 
83  // Construct block matrix - this uses the boost::range 'transformed' construct to apply
84  // internal::getColsJF (defined just above here in this file) to each term. This
85  // transforms the list of terms into a list of variable dimensions, by extracting the
86  // number of columns in each matrix. This is done to avoid separately allocating an
87  // array of dimensions before constructing the VerticalBlockMatrix.
88  Ab_ = VerticalBlockMatrix(terms | transformed(&internal::getColsJF), b.size(), true);
89 
90  // Check and add terms
91  DenseIndex i = 0; // For block index
92  for(typename TERMS::const_iterator termIt = terms.begin(); termIt != terms.end(); ++termIt) {
93  const std::pair<Key, Matrix>& term = *termIt;
94 
95  // Check block rows
96  if(term.second.rows() != Ab_.rows())
97  throw InvalidMatrixBlock(Ab_.rows(), term.second.rows());
98 
99  // Assign key and matrix
100  Base::keys_[i] = term.first;
101  Ab_(i) = term.second;
102 
103  // Increment block index
104  ++ i;
105  }
106 
107  // Assign RHS vector
108  getb() = b;
109 
110  // Assign noise model
111  model_ = noiseModel;
112  }
