227  template<typename ValueType>
228  const ValueType& Values::at(Key j) const {
229  // Find the item
230  KeyValueMap::const_iterator item = values_.find(j);
231 
232  // Throw exception if it does not exist
233  if(item == values_.end())
234  throw ValuesKeyDoesNotExist("retrieve", j);
235 
236  // Check the type and throw exception if incorrect
237  if(typeid(*item->second) != typeid(ValueType))
238  throw ValuesIncorrectType(j, typeid(*item->second), typeid(ValueType));
239 
240  // We have already checked the type, so do a "blind" static_cast, not dynamic_cast
241  return static_cast<const ValueType&>(*item->second);
242  }

255  // We have already checked the type, so do a "blind" static_cast, not dynamic_cast
