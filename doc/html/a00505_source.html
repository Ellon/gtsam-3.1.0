147  /*
148  * Vector of errors returns 2D vector
149  * @param E essential matrix
150  * @param d inverse depth d
151  */
152  Vector evaluateError(const EssentialMatrix& E, const LieScalar& d,
153  boost::optional<Matrix&> DE = boost::none, boost::optional<Matrix&> Dd =
154  boost::none) const {
155 
156  // We have point x,y in image 1
157  // Given a depth Z, the corresponding 3D point P1 = Z*(x,y,1) = (x,y,1)/d
158  // We then convert to second camera by P2 = 1R2'*(P1-1T2)
159  // The homogeneous coordinates of can be written as
160  // 2R1*(P1-1T2) == 2R1*d*(P1-1T2) == 2R1*((x,y,1)-d*1T2)
161  // where we multiplied with d which yields equivalent homogeneous coordinates.
162  // Note that this is just the homography 2R1 for d==0
163  // The point d*P1 = (x,y,1) is computed in constructor as dP1_
164 
165  // Project to normalized image coordinates, then uncalibrate
166  Point2 pn;
167  if (!DE && !Dd) {
168 
169  Point3 _1T2 = E.direction().point3();
170  Point3 d1T2 = d * _1T2;
171  Point3 dP2 = E.rotation().unrotate(dP1_ - d1T2); // 2R1*((x,y,1)-d*1T2)
172  pn = SimpleCamera::project_to_camera(dP2);
173 
174  } else {
175 
176  // Calculate derivatives. TODO if slow: optimize with Mathematica
177  // 3*2 3*3 3*3 2*3
178  Matrix D_1T2_dir, DdP2_rot, DP2_point, Dpn_dP2;
179 
180  Point3 _1T2 = E.direction().point3(D_1T2_dir);
181  Point3 d1T2 = d * _1T2;
182  Point3 dP2 = E.rotation().unrotate(dP1_ - d1T2, DdP2_rot, DP2_point);
183  pn = SimpleCamera::project_to_camera(dP2, Dpn_dP2);
184 
185  if (DE) {
186  Matrix DdP2_E(3, 5);
187  DdP2_E << DdP2_rot, -DP2_point * d * D_1T2_dir; // (3*3), (3*3) * (3*2)
188  *DE = f_ * Dpn_dP2 * DdP2_E; // (2*3) * (3*5)
189  }
190 
191  if (Dd) // efficient backwards computation:
192  // (2*3) * (3*3) * (3*1)
193  *Dd = -f_ * (Dpn_dP2 * (DP2_point * _1T2.vector()));
194 
195  }
196  Point2 reprojectionError = pn - pn_;
197  return f_ * reprojectionError.vector();
198  }

261  boost::optional<Matrix&> DE = boost::none, boost::optional<Matrix&> Dd =

271  EssentialMatrix cameraE = E.rotate(cRb_, D_cameraE_E);
272  Vector e = Base::evaluateError(cameraE, d, D_e_cameraE, Dd);
273  *DE = D_e_cameraE * D_cameraE_E; // (2*5) * (5*5)
